// Abstract Plasma Shader (Fragment Shader) #version 330 소 precision highp float; uniform float u_time; uniform vec2 u_resolution; uniform sampler2D u_texture; // Optional: for sampling distortion or background // ================================================================== // Utility Functions (Noise Generation) // A standard 2D noise function (e.g., based on pseudo-random hash) // Using a simplified hash approach for demonstration. // ================================================================== float hash(vec2 p) { return fract(sin(dot(p, vec2(41.33, 2.0))) * 1234.56789); } float noise(vec2 p) { vec2 i = floor(p); vec2 f = fract(p); // Smooth interpolation (smootherstep) vec2 u = f * f * (3.0 - 2.0 * f); float a = hash(i + vec2(0.0, 0.0)); float b = hash(i + vec2(1.0, 0.0)); float c = hash(i + vec2(0.0, 1.0)); float d = hash(i + vec2(1.0, 1.0)); // Bilinear interpolation return mix(mix(a, b, u.x), mix(c, d, u.x), u.y); } // Combines multiple octaves of noise for fractal detail float fbm(vec2 p, int octaves, float lacunarity, float persistence) { float total = 0.0; float amplitude = 1.0; float frequency = 1.0; for (int i = 0; i < octaves; i++) { total += noise(p * frequency) * amplitude; amplitude *= persistence; frequency *= lacunarity; } return total; } // ================================================================== // Main Shader Function // ================================================================== void main() { // Normalized coordinates vec2 uv = gl_FragCoord.xy / u_resolution.xy; // Center coordinates for toroidal effect vec2 center_uv = uv * 2.0 - 1.0; // Time-space coordinates for movement vec2 p = center_uv * 3.0 + vec2(u_time * 0.1, u_time * 0.05); // 1. Base Turbulence (FBM) float noise_val = fbm(p * 0.8, 6, 2.0, 0.5); // 2. Second Layer of Movement/Detail // Use a different offset direction for secondary movement float secondary_noise = fbm(p * 0.5 + vec2(u_time * 0.05, u_time * 0.1), 4, 2.5, 0.4); // Combine and scale the noise into a plasma energy field float plasma_intensity = (noise_val * 0.5 + secondary_noise * 0.5); // Normalize and smooth the intensity (0.0 to 1.0 range) float energy = smoothstep(0.0, 1.0, pow(plasma_intensity, 0.8) * 0.8 + 0.2); // 3. Plasma Color Mapping (Iridescence/Energy Shift) // Use sine/cosine functions on the noise/time for cyclic color shifts float color_shift = sin(p.x * 0.5 + u_time * 0.2) * 0.5 + 0.5; float color_depth = sin(p.y * 0.5 + u_time * 0.1) * 0.5 + 0.5; // Create a highly saturated, shifting color palette (e.g., deep blues/purples/cyan) vec3 color_a = vec3(0.5 + 0.5 * cos(u_time * 0.1 + p.x * 0.3), 0.2 + 0.8 * sin(u_time * 0.05 + p.y * 0.3), 1.0); // Blue/Purple base vec3 color_b = vec3(0.8, 0.5 + 0.5 * cos(u_time * 0.02), 0.2); // Secondary glow/red hint // Interpolate between two color states based on energy vec3 plasma_color = mix(color_a, color_b, energy * 0.8); // 4. Final Composition and Glow // The glow effect: Multiply the base color by the energy map and boost the overall brightness vec3 final_color = plasma_color * energy * 1.5; // Add a slight emission/ambient glow to the edges final_color += vec3(0.05) * (1.0 - smoothstep(0.0, 0.5, length(uv - vec2(0.5)))); // Tone mapping and clamping for vibrant look final_color = pow(final_color, vec3(0.8)); gl_FragColor = vec4(final_color, 1.0); }
