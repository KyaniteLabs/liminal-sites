#version 300 es precision highp float; uniform vec2 u_resolution; uniform float u_time; out vec4 fragColor; // Hash function for pseudo-random values float random(vec2 co) { return fract(sin(dot(co.xy, vec2(12.9898, 78.233))) * 43758.5453); } // Perlin-like noise function (simple implementation for effect) float noise(vec2 p) { vec2 i = floor(p); vec2 f = fract(p); vec2 u = f * f * (3.0 - 2.0 * f); float a = random(i); float b = random(i + vec2(1.0, 0.0)); float c = random(i + vec2(0.0, 1.0)); float d = random(i + vec2(1.0, 1.0)); return mix(mix(a, b, u.x), mix(c, d, u.x), u.y); } void main() { vec2 uv = gl_FragCoord.xy / u_resolution.xy; vec2 p = uv * 5.0; // Scale coordinates for pattern detail // Time-dependent offset for animation float time_factor = u_time * 0.8; p.x += time_factor * 0.2; p.y += time_factor * 0.1; // 1. Wave/Swirl Component (Using sine waves for primary structure) float wave1 = sin(p.x * 5.0 + u_time * 1.5) * 0.3; float wave2 = cos(p.y * 4.0 - u_time * 1.0) * 0.3; float swirl = sin(dot(p, vec2(1.0, 0.8)) * 4.0 + u_time * 2.0) * 0.2; float plasma_base = abs(wave1) + abs(wave2) + abs(swirl); plasma_base = pow(plasma_base, 1.5) * 0.5; // 2. Turbulence/Noise Layer (Adding flickering detail) float noise_scale = 1.5; float noise_detail = noise(p * noise_scale + u_time * 0.5) * 0.5; // 3. Combining effects (Plasma glow effect) // Use the noise to modulate the intensity of the base pattern float intensity = plasma_base + noise_detail; // Curve the intensity for a more defined glow/plasma look float final_intensity = smoothstep(0.1, 1.2, intensity) * 1.5; // Color mapping: Plasma colors (blues, purples, electric greens) vec3 color_plasma = vec3( 0.5 + 0.2 * sin(p.y * 2.0 + u_time * 0.8), // Red channel variation 0.3 + 0.4 * cos(p.x * 2.0 + u_time * 0.9), // Green channel variation 0.8 + 0.2 * sin(p.x * 3.0 + u_time * 1.1) // Blue channel variation ); // Final color: Multiply the structural color by the intensity (glow effect) vec3 final_color = color_plasma * final_intensity; // Add a slight white/bright core effect for high energy zones final_color += pow(final_intensity, 2.0) * 0.5; gl_FragColor = vec4(final_color, 1.0); }
