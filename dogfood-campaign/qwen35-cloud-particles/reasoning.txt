let particles = [];

function setup() {
  createCanvas(windowWidth, windowHeight);
  colorMode(HSB, 360, 100, 100, 100);
}

function draw() {
  background(0);
  // Emit particles
  for(let i=0; i<5; i++) {
    particles.push(new Particle(mouseX, mouseY));
  }
  // Update/Draw
  for (let i = particles.length - 1; i >= 0; i--) {
    let p = particles[i];
    p.update();
    p.show();
    if (p.isDead()) {
      particles.splice(i, 1);
    }
  }
}

class Particle {
  constructor(x, y) {
    this.pos = createVector(x, y);
    this.vel = p5.Vector.random2D().mult(random(2, 5)); // Explosion
    this.acc = createVector(0, 0.2); // Gravity
    this.hue = random(360);
    this.lifespan = 255;
  }
  update() {
    this.vel.add(this.acc);
    this.pos.add(this.vel);
    this.lifespan -= 5;
  }
  show() {
    stroke(this.hue, 80, 100, this.lifespan);
    strokeWeight(2);
    point(this.pos.x, this.pos.y);
  }
  isDead() {
    return this.lifespan < 0;
  }
}
